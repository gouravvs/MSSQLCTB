4. Now connected lets document the basic Linux commands 
    ## FILE & DIRECTORY NAVIGATION
            pwd            # Show current directory path
            mkdir Test     # Create directory
            ls             # List files and directories
            ls -l          # Detailed file listing
            ls -a          # Show hidden files
            cd /path       # Change directory
            cd ..          # Move one directory up
            cd ~           # Go to home directory
            tree           # Show directory structure (if installed)

           -- ### Optional ### --
            mkdir -p linux_lab/{docs,scripts,logs} # create multiple directories inside one directory

    ## FILE OPERATIONS
            touch file     # Create empty file
            cat file       # Display file content
            less file      # View file page by page
            more file      # View file (basic pager)
            cp src dst     # Copy file
            mv src dst     # Move or rename file
            rm file        # Delete file
            rm -r dir      # Delete directory recursively
            stat file      # Show file metadata
            file file      # Show file type
            vi file        # To edit a text File
           vi intro.txt

           WORKFLOW for editing and saving a File 
                                1.    i
                                2.  (type your text)
                                3. ESC
                                4. :wq
                                5. ENTER
                        Use :q! to exit without saving AND NOT USE crtl+C 

    ## SEARCH & FIND
            find / -name filename # Find file by name over all 
            find . -name filename # Find file by name in the current directory 
            find <where_to_search> <criteria>

            locate file    # Fast search (uses index) ### Before that make sure to update database by updatedb command ###
            grep <options> <text> <where_to_search>

            grep -ri text . # Case-insensitive search in the present directory 
            grep -ri text DirectoryName # Case-insensitive search in the that directory 
            grep -r text . # Recursive search in present directory 

                                grep -r Linux .
                                # searches in linux_lab and below

                                find . -type f
                                # lists all files under linux_lab

                                find . -type d
                                # lists all directories

    ## FILE SIZE & DISK
            df -h          # Disk space usage
            du -sh dir     # Directory size
            ls -lh         # File sizes in readable form

    ## MEMORY & CPU
            free -h        # Memory usage
            top            # Live CPU/memory usage
            htop           # Enhanced top (if installed)
            uptime         # System running time

    ## PROCESS MANAGEMENT
            ps -ef         # List all processes
            ps aux         # Process snapshot
            top            # Real-time process view
            kill PID       # Kill process
            kill -9 PID    # Force kill
            pkill name     # Kill by process name

    ## NETWORKING
            ip a            # Show IP addresses
            ip r            # Show routing table
            ping host       # Test connectivity
            ss -tulpn       # Listening ports
            netstat -tulpn  # Legacy port view
            hostname        # Show hostname
            hostnamectl    # Host/system info

    ## USERS & PERMISSIONS
            whoami          # Current user
            id              # User and group IDs
            who              # Logged-in users
            groups          # User groups
            chmod 755 file  # Change permissions
            chown user file # Change file owner

    ## FILE REDIRECTION & PIPE (VERY IMPORTANT)
            cmd > file       # Output to file
            cmd >> file      # Append output
            cmd < file       # Input from file
            cmd1 | cmd2      # Pipe output

     ## LOGS & TROUBLESHOOTING
        journalctl           # System logs
        journalctl -xe       # Error logs
        journalctl -u svc    # Service-specific logs
        dmesg                # Kernel messages

     ## OS-VERSION 
        cat /etc/os-release  # Version of the OS
        uname -r             # Display Kernel Version
